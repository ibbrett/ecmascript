```js
const list = ['apple', 'cherry', 'pumpkin', 'pecan', 'blueberry', 'peach'];
const entriesIterator = list.entries();

let result = entriesIterator.next();
while (!result.done) {
  console.log(result.value);
  result = entriesIterator.next();
}

// Expected output:
// [0, 'apple']
// [1, 'cherry']
// [2, 'pumpkin']
// [3, 'pecan']
// [4, 'blueberry']
// [5, 'peach']
// {value: undefined, done: true}
```
