```js
const list = ['apple', 'cherry', 'pumpkin', 'pecan', 'blueberry', 'peach'];
const entriesIterator = list.keys();

for (const key of entriesIterator) {
  console.log('entry key: ', key);
}

// Expected output:
// entry key:  0
// entry key:  1
// entry key:  2
// entry key:  3
// entry key:  4
// entry key:  5
```
