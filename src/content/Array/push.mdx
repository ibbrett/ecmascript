- to add item:
  - to index of array use [splice()](../splice-add-item/)
  - to the beginning of array use [unshift()](../unshift/)

<div class="copy-to-clipboard-wrap" id="code-sample-01">
<CopyToClipboard id="code-sample-01"></CopyToClipboard>
```js
const list = ['apple', 'banana', 'cantaloupe'];
console.log('initial list', list);
// Expected output: ['apple', 'banana', 'cantaloupe']

// get/remove first item
const returnValue = list.push('lemon');

console.log('add item to end of list', list);
// Expected output: ['apple', 'banana', 'cantaloupe', 'lemon']

console.log('list length', returnValue);
// Expected output: 4
```
