const list = ['apple', 'cherry', 'pumpkin', 'pecan', 'blueberry', 'peach'];
console.log('all items', list);
// Expected output: 

const fiveLetterItems = list.filter( item => item.length === 5 );
console.log('five letter items', fiveLetterItems);
// Expected output:
