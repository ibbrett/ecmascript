```js
const list = ['apple', 'cherry', 'pumpkin', 'pecan', 'blueberry', 'peach'];
const entriesIterator = list.values();

for (const value of entriesIterator) {
  console.log('entry value: ', value);
}

// Expected output:
// entry key:  apple
// entry key:  cherry
// entry key:  pumpkin
// entry key:  pecan
// entry key:  blueberry
// entry key:  peach
```
