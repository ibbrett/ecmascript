```js
const list = ['apple', 'banana', 'cantaloupe', 'orange', 'lemon', 'pear'];

console.log('initial list', list);
// Expected output: ['apple', 'banana', 'cantaloupe', 'orange', 'lemon', 'pear']

const listCopy = list.slice();
console.log('slice whole list', listCopy);
// Expected output: ['apple', 'banana', 'cantaloupe', 'orange', 'lemon', 'pear']

console.log('slice 3nd to 3rd-to-last section of list', list.slice(2, -2));
// Expected output: ['cantaloupe', 'orange']

listCopy.length = 2;
console.log('truncate length to 2', listCopy);
// Expected output: ['apple', 'banana']

console.log('original list has not changed', list);
// Expected output: ['apple', 'banana', 'cantaloupe', 'orange', 'lemon', 'pear']
```
