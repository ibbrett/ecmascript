<div class="copy-to-clipboard-wrap" id="code-sample-01">
<CopyToClipboard id="code-sample-01"></CopyToClipboard>
```js
const list = ['apple', 'cherry', 'pumpkin', 'pecan', 'blueberry', 'peach'];
console.log('initial list', list);
// Expected output: ['peach', 'cherry', 'pumpkin', 'pecan', 'blueberry', 'peach']

// copy last item to index 0
console.log(list.copyWithin(0, -1));
// Expected output: ['peach', 'cherry', 'pumpkin', 'pecan', 'blueberry', 'peach']

// copy 5th item to index 1
console.log(list.copyWithin(1, 4, 5));
// ['peach', 'blueberry', 'pumpkin', 'pecan', 'blueberry', 'peach']

// copy items to index 0 from index 3 to end of array
console.log(list.copyWithin(1, 3));
// Expected output: ['peach', 'pecan', 'blueberry', 'peach', 'blueberry', 'peach']
```
