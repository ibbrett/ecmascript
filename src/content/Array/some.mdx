```js
const list = ['apple', 'cherry', 'pumpkin', 'pecan', 'blueberry', 'peach'];
console.log('all items', list);
// Expected output: 

let minLength = 9;
console.log(`has some items with length ${minLength}`, list.some( item => item.length >= minLength ));
// Expected output: true

minLength = 10;
console.log(`has some items with length ${minLength}`, list.some( item => item.length >= minLength ));
// Expected output: false
```
