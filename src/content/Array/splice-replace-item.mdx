```js
const list = ['apple', 'cantaloupe'];
console.log('initial list', list);
// Expected output: ['apple', 'cantaloupe']

// replace item at a particular index
let returnValue = list.splice(1, 1, 'banana');

console.log('replace item', list);
// Expected output: ['apple', 'banana']

console.log('replaced value', returnValue);
// Expected output: cantaloupe

// alternative methods of item replacement  
list[1] = 'orange';

console.log('replace item', list);
// Expected output: ['apple', 'orange']

list.at(1) = 'lemon';

console.log('replace item', list);
// Expected output: ['apple', 'lemon']
```
