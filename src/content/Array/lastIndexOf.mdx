```js
const list = ['apple', 'blueberry', 'cherry', 'apple'];
console.log('all items', list);
// Expected output: 

let searchItem = 'apple';
console.log(`last index for ${searchItem} in list`, list.lastIndexOf(searchItem));
// Expected output: 3

searchItem = 'banana';
console.log(`last index for ${searchItem} in list`, list.lastIndexOf(searchItem));
// Expected output: -1
```
