```js
const list = ['apple', 'banana', 'cantaloupe', 'orange', 'lemon', 'pear'];

console.log('list', list);
// Expected output: ['apple', 'banana', 'cantaloupe', 'orange', 'lemon', 'pear']

console.log('typeof list', typeof list);
// Expected output: typeof list object

console.log('list is an array?', Array.isArray(list));
// Expected output: list is an array? true
```
