{frontmatter.what}

- to add item:
  - at index of array use [splice()](../splice-add-item/)
  - to the end of array use [push()](../push/)

<div class="copy-to-clipboard-wrap" id="code-sample-01">
<CopyToClipboard id="code-sample-01"></CopyToClipboard>
```js
const list = ['apple', 'cantaloupe'];
console.log('initial list', list);
// Expected output: ['apple', 'cantaloupe']

// prepend item
const returnValue = list.unshift('banana');

console.log('prepend item', list);
// Expected output: ['banana', 'apple', 'cantaloupe']

console.log('returnValue', returnValue);
// Expected output: 3
```
</div>

- to add items:

<div class="copy-to-clipboard-wrap" id="code-sample-02">
<CopyToClipboard id="code-sample-02"></CopyToClipboard>
```js
const list = ['apple', 'cantaloupe'];
console.log('initial list', list);
// Expected output: ['apple', 'cantaloupe']

// prepend individuals items
const returnValue = list.unshift('banana', 'orange', 'lemon');

console.log('prepend items', list);
// Expected output: ['banana', 'orange', 'lemon', 'apple', 'cantaloupe']

console.log('returnValue', returnValue);
// Expected output: 5

// prepend array of items
list.unshift(...['dragon fruit', 'guava']);
console.log('prepend items', list);
// Expected output: ['dragon fruit', 'guava', 'banana', 'orange', 'lemon', 'apple', 'cantaloupe']
```
</div>
