{frontmatter.what}

- to remove item from the beginning of array use [shift()](../shift/)
- to remove item from the end of array use [pop()](../pop/)

<div class="copy-to-clipboard-wrap" id="code-sample-01">
<CopyToClipboard id="code-sample-01"></CopyToClipboard>
```js
const list = ['apple', 'banana', 'cantaloupe', 'orange', 'lemon', 'dragon fruit', 'pear'];
console.log('initial list', list);
// Expected output: ['apple', 'banana', 'cantaloupe', 'orange', 'lemon', 'dragon fruit', 'pear']

// remove an item at a particular index
let returnValue = list.splice(1, 1);

console.log('remove item from list', list);
// Expected output: ['apple', 'cantaloupe', 'orange', 'lemon', 'dragon fruit', 'pear']

console.log('removed item', returnValue);
// Expected output: ['banana']

// remove multiple items starting at a particular index
returnValue = list.splice(2, 3);

console.log('remove items from list', list);
// Expected output: ['apple', 'cantaloupe', 'pear']

console.log('removed items', returnValue);
// Expected output: ['orange', 'lemon', 'dragon fruit']
```
