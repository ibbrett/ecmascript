- simplifies the process of creating arrays with specified elements
- create arrays with a dynamic number of elements

const getArrayFromArgs = ( ...args ) => Array.of( ...args );

const dynamicArray = getArrayFromArgs( true, 'carpe diem', 42, {} );

console.log(dynamicArray);
