<div class="copy-to-clipboard-wrap" id="code-sample-01">
<CopyToClipboard id="code-sample-01"></CopyToClipboard>
```js
const fruits = ['apple', 'banana', 'cantaloupe'];
const vegetables = ['asparagus', 'broccoli', 'carrot'];

console.log('fruits and vegetables', fruits.concat(vegetables));
// Expected output: ['apple', 'banana', 'cantaloupe', 'asparagus', 'broccoli', 'carrot']
```
</div>

alternative: use the spread operator

<div class="copy-to-clipboard-wrap" id="code-sample-02">
<CopyToClipboard id="code-sample-02"></CopyToClipboard>
```js
const fruits = ['apple', 'banana', 'cantaloupe'];
const vegetables = ['asparagus', 'broccoli', 'carrot'];

console.log('concat', [].concat(fruits,vegetables));
console.log('spread', [...fruits,...vegetables]);
// Expected output: ['apple', 'banana', 'cantaloupe', 'asparagus', 'broccoli', 'carrot']
```
</div>
